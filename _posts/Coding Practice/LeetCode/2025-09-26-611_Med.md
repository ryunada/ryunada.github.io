nums = [2,2,3,4]  # Output: 3

from typing import List

class Solution:
    def triangleNumber(self, nums: List[int]) -> int:
        # ì˜¤ë¦„ì°¨ìˆœ ì •ë ¬
        nums.sort()
        answer = 0

        # ì‚¼ê°í˜• ì¡°ê±´ : a + b > c
        for k in range(len(nums) - 1, 1, -1):
            i, j = 0, k - 1
            while i < j:
                if nums[i] + nums[j] > nums[k]:
                    answer += j - i
                    j -= 1
                else:
                    i += 1
        return answer
        
        

S = Solution()
print(S.triangleNumber(nums))
